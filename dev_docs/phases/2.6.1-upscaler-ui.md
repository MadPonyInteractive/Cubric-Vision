```js
promptBox = new PromptBox({

    onImageDrop: async (item) => { ... }
```

**Replace with:**
```js

if (denoiseContainer) {
    denoiseContainer.innerHTML = '';
    denoiseSlider = MpiSlider.mount(denoiseContainer, {
        label: 'DENOISE',
        min: 0, max: 1, step: 0.01,
        value: state.toolComfySettings?.upscaler?.denoise || 0.3,
```
