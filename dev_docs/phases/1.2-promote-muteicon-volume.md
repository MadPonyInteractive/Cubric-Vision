### 1. Create `js/components/Compounds/MpiMuteIcon/MpiMuteIcon.js`

This component is a specialized button that cycles through mute/volume-low/volume-high states.

```js
/**
 * MpiMuteIcon — Compound component for mute/volume toggle.
 * Composes MpiIconButton with volume-state-aware icon swap.
 *
 * Props: MpiMuteIconProps
 * Emits: 'change' { muted: boolean }
 */
import { ComponentFactory } from '../../factory.js';
import { MpiIconButton } from '../MpiIconButton/MpiIconButton.js';

export const MpiMuteIcon = ComponentFactory.create({
    name: 'MpiMuteIcon',
    css: ['js/components/Compounds/MpiMuteIcon/MpiMuteIcon.css'],
    template: (props) => {
        const icon = props.muted ? 'volumeOff'
            : props.volume > 0.5 ? 'volumeHigh' : 'volumeLow';
        return MpiIconButton.template({ icon, size: props.size || 'md', info: props.muted ? 'Unmute' : 'Mute' });
    },
    setup: (el, props, emit) => {
        let muted = props.muted ?? false;
        let volume = props.volume ?? 1.0;

        function _updateIcon() {
            const icon = el.querySelector('[data-icon]') || el.querySelector('svg');
            // Re-render icon inline based on state
            const name = muted ? 'volumeOff' : volume > 0.5 ? 'volumeHigh' : 'volumeLow';
            // Use MpiIconButton's setup to re-render — or simply re-mount.
            // Simplest: dispatch state to the icon element class attribute.
            el.dataset.muted = muted ? '1' : '0';
        }

        el.addEventListener('click', () => {
            muted = !muted;
            _updateIcon();
            emit('change', { muted, volume });
        });

        /** Public API: update volume level (changes icon without toggling mute) */
        el._setVolume = (v) => { volume = v; _updateIcon(); };
        /** Public API: set muted state programmatically */
        el._setMuted = (m) => { muted = m; _updateIcon(); };
    }
});
```

> **Note:** The exact icon-swap implementation should match the pattern in `MpiIconButton.js`. Read that file first and reuse its technique. The above is a structural reference — adapt to fit the actual factory API.

### 3. Create `js/components/Compounds/MpiVolumeControl/MpiVolumeControl.js`

This orchestrates `MpiMuteIcon` + `MpiSlider` in a popup, replacing the old `VolumeControl.js`.

Key behaviors from the old component (get from `git show HEAD~1:js/components/VolumeControl.js`):
- Mute icon click toggles mute, emits `onChange(volume, muted)`
- Slider communicates volume `0–1`
- When muted: slider shows 0 visually but remembers previous volume
- On unmute: slider restores previous volume
- Popup appears on hover over the mute icon

```js
/**
 * MpiVolumeControl — Compound: MpiMuteIcon + MpiSlider popup.
 * Props: MpiVolumeControlProps
 * Emits: 'change' { volume: number, muted: boolean }
 */
import { ComponentFactory } from '../../factory.js';
import { MpiMuteIcon }  from '../MpiMuteIcon/MpiMuteIcon.js';
import { MpiSlider }    from '../MpiSlider/MpiSlider.js';
import { MpiPopup }     from '../../Primitives/MpiPopup/MpiPopup.js';

export const MpiVolumeControl = ComponentFactory.create({
    name: 'MpiVolumeControl',
    css: ['js/components/Compounds/MpiVolumeControl/MpiVolumeControl.css'],
    template: () => `<div class="mpi-volume-control"></div>`,
    setup: (el, props, emit) => {
        let volume = props.volume ?? 1.0;
        let muted  = props.muted  ?? false;
        let _prevVolume = volume;

        // Mount sub-components
        const iconSlot   = document.createElement('div');
        const popupSlot  = document.createElement('div');
        el.appendChild(iconSlot);
        el.appendChild(popupSlot);

        const muteIcon = MpiMuteIcon.mount(iconSlot, { volume, muted });
        const popup    = MpiPopup.mount(popupSlot, { id: `vol-popup-${Math.random().toString(36).slice(2)}` });
        // Mount slider inside popup
        const sliderSlot = document.createElement('div');
        popup.el.appendChild(sliderSlot);
        const slider = MpiSlider.mount(sliderSlot, {
            min: 0, max: 100, step: 1, value: muted ? 0 : Math.round(volume * 100),
            wheel: true
        });

        // Show/hide popup on hover
        iconSlot.addEventListener('mouseenter', () => popup.el.classList.add('active'));
        el.addEventListener('mouseleave', () => popup.el.classList.remove('active'));

        // Mute toggle
        muteIcon.on('change', ({ muted: m }) => {
            muted = m;
            if (muted) { _prevVolume = volume; slider.el.querySelector('input').value = 0; }
            else { volume = _prevVolume; slider.el.querySelector('input').value = Math.round(volume * 100); }
            emit('change', { volume: muted ? 0 : volume, muted });
        });

        // Slider change
        slider.on('change', ({ value }) => {
            volume = value / 100;
            muted = volume === 0;
            muteIcon.el._setVolume?.(volume);
            muteIcon.el._setMuted?.(muted);
            emit('change', { volume, muted });
        });
    }
});
```

> **Important:** Read the actual `MpiSlider` and `MpiPopup` source before writing this — the above shows the intent. Adapt the event APIs to match what those components actually emit.
